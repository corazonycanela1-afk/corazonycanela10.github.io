<script>
function cambiarCantidad(id, valor){
    let input = document.getElementById(id);
    let cantidad = parseInt(input.value);

    if(isNaN(cantidad)){ cantidad = 0; }

    cantidad += valor;

    if(cantidad < 0){ cantidad = 0; }

    input.value = cantidad;
}
